---
title: "Get User"
api: "GET /users/{id}"
description: "Retrieve a specific user by their ID"
---

## Authentication

This endpoint requires HTTP Basic Authentication.

## Path Parameters

<ParamField path="id" type="long" required>
  The unique identifier of the user to retrieve
</ParamField>

## Response

<ResponseField name="id" type="long">
  Unique identifier for the user
</ResponseField>

<ResponseField name="firstname" type="string">
  The user's first name
</ResponseField>

<ResponseField name="lastname" type="string">
  The user's last name
</ResponseField>

<ResponseField name="email" type="string">
  The user's email address
</ResponseField>

<ResponseField name="borrowedBooks" type="array">
  Set of rental records for books borrowed by this user
  <Expandable title="RentalDto properties">
    <ResponseField name="id" type="long">
      Unique identifier for the rental
    </ResponseField>
    <ResponseField name="user" type="object">
      User entity reference
    </ResponseField>
    <ResponseField name="book" type="object">
      Book entity reference
    </ResponseField>
    <ResponseField name="loanDate" type="string">
      ISO 8601 datetime when the book was borrowed
    </ResponseField>
    <ResponseField name="returnDate" type="string">
      ISO 8601 datetime when the book should be returned
    </ResponseField>
    <ResponseField name="returned" type="boolean">
      Whether the book has been returned
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="googleId" type="string">
  Google OAuth ID (only included if set)
</ResponseField>

<ResponseField name="displayPicture" type="string">
  URL to the user's display picture (only included if set)
</ResponseField>

## Status Codes

<ResponseField name="200" type="OK">
  User successfully retrieved
</ResponseField>

<ResponseField name="401" type="Unauthorized">
  Missing or invalid authentication credentials
</ResponseField>

<ResponseField name="404" type="Not Found">
  User with the specified ID does not exist
</ResponseField>
